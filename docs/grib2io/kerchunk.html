grib2io.kerchunk

Kerchunk Reference Manifest Generator

Provides ReferenceGenerator, which scans one or more GRIB2 files using grib2io's build_index() infrastructure and produces a Kerchunk v1 reference manifest mapping Zarr chunk keys to [url, offset, length] tuples within the original files.

The manifest can be serialized to JSON or Parquet and later opened with fsspec.filesystem("reference") to create a virtual Zarr store that reads data lazily from the original GRIB2 bytes, decoded on-the-fly by grib2io.codecs.Grib2Codec.

Example
>>> from grib2io.kerchunk import ReferenceGenerator
>>> gen = ReferenceGenerator("gfs.grib2")
>>> manifest = gen.generate()
>>> gen.to_json("gfs_refs.json")

 82class ReferenceGenerator:
 83    """Generate Kerchunk v1 reference manifests from GRIB2 files.
 84
 85    Parameters
 86    ----------
 87    file_paths : str or list of str
 88        One or more GRIB2 file paths (local paths or URIs).
 89    filters : dict, optional
 90        Filter GRIB2 messages by metadata attributes.  Keys can be any
 91        ``Grib2Message`` attribute name (e.g. ``shortName``, ``leadTime``).
 92    """
 93
 94    def __init__(
 95        self,
 96        file_paths: Union[str, List[str]],
 97        filters: Optional[Dict[str, Any]] = None,
 98    ):
 99        _ensure_numcodecs()
100
101        if isinstance(file_paths, (str, os.PathLike)):
102            file_paths = [str(file_paths)]
103        else:
104            file_paths = [str(p) for p in file_paths]
105
106        # Validate file accessibility
107        for fp in file_paths:
108            if not os.path.isfile(fp):
109                raise FileNotFoundError(f"GRIB2 file not found: {fp}")
110
111        self.file_paths = file_paths
112        self.filters = filters or {}
113        self._manifest: Optional[dict] = None
114
115    def generate(self) -> dict:
116        """Scan files and produce a Kerchunk v1 reference manifest.
117
118        Returns
119        -------
120        dict
121            Kerchunk reference spec v1 dict with keys ``"version"`` and
122            ``"refs"``.
123        """
124        refs: Dict[str, Any] = {}
125
126        # .zgroup at root
127        refs[".zgroup"] = json.dumps({"zarr_format": 2})
128
129        # Collect all messages across files, keyed by variable group
130        # group_key = (shortName, typeOfFirstFixedSurface, pdtn, typeOfSecondFixedSurface)
131        # This ensures messages with different surface types are not mixed.
132        all_var_messages: Dict[tuple, list] = {}
133
134        for file_path in self.file_paths:
135            file_uri = _file_uri(file_path)
136            try:
137                self._scan_file(file_path, file_uri, all_var_messages)
138            except Exception as e:
139                raise ValueError(f"Failed to parse GRIB2 file '{file_path}': {e}") from e
140
141        # For each variable group, map messages to dimensions and build refs.
142        # Track used variable names to handle collisions (same shortName
143        # but different surface types).
144        used_var_names: Dict[str, int] = {}
145        for group_key, msg_entries in all_var_messages.items():
146            var_name = group_key[0]  # shortName is the first element
147            if var_name in used_var_names:
148                used_var_names[var_name] += 1
149                zarr_var_name = f"{var_name}_{used_var_names[var_name]}"
150            else:
151                used_var_names[var_name] = 0
152                zarr_var_name = var_name
153            self._build_variable_refs(zarr_var_name, msg_entries, refs)
154
155        self._manifest = {"version": 1, "refs": refs}
156        return self._manifest
157
158    def to_json(self, output_path: str) -> None:
159        """Serialize the manifest to a JSON file.
160
161        Parameters
162        ----------
163        output_path : str
164            Path to the output JSON file.
165        """
166        if self._manifest is None:
167            self.generate()
168        with open(output_path, "w") as f:
169            json.dump(self._manifest, f)
170
171    def to_parquet(self, output_path: str) -> None:
172        """Serialize the manifest to a Parquet reference store.
173
174        Parameters
175        ----------
176        output_path : str
177            Path to the output Parquet directory.
178        """
179        _ensure_kerchunk()
180        if self._manifest is None:
181            self.generate()
182        from kerchunk.df import refs_to_dataframe
183
184        refs_to_dataframe(self._manifest, output_path)

190    def _scan_file(
191        self,
192        file_path: str,
193        file_uri: str,
194        all_var_messages: Dict[str, list],
195    ) -> None:
196        """Scan a single GRIB2 file and collect message entries."""
197        with grib2io.open(file_path, save_index=False, use_index=False) as f:
198            index = f._index
199            msgs = list(f)
200
201        n_msgs = len(msgs)
202        for i in range(n_msgs):
203            msg = msgs[i]
204
205            # Apply filters
206            if not self._matches_filters(msg):
207                continue
208
209            sec_offsets = index["sectionOffset"][i]
210            sec_sizes = index["sectionSize"][i]
211            bmapflag = index["bmapflag"][i]
212
213            # Section 7 offset and length
214            sec7_offset = sec_offsets[7]
215            sec7_length = sec_sizes[7]
216
217            # Section 6 (bitmap) offset and length
218            sec6_offset = sec_offsets[6] if bmapflag in {0, 254} else None
219            sec6_length = sec_sizes[6] if bmapflag in {0, 254} else None
220
221            # Build a composite variable key that includes the surface type
222            # to avoid grouping messages with different surface types together.
223            # This mirrors how the xarray backend requires filtering to a
224            # single typeOfFirstFixedSurface.
225            var_name = str(msg.shortName)
226            type_of_first_fixed_surface = msg.typeOfFirstFixedSurface
227            if hasattr(type_of_first_fixed_surface, "value"):
228                toffs_val = type_of_first_fixed_surface.value
229            else:
230                toffs_val = type_of_first_fixed_surface
231
232            # Also include typeOfGeneratingProcess and
233            # productDefinitionTemplateNumber to disambiguate further
234            # (same approach as xarray backend's required_uniques)
235            pdtn = msg.productDefinitionTemplateNumber
236            if hasattr(pdtn, "value"):
237                pdtn_val = pdtn.value
238            else:
239                pdtn_val = pdtn
240
241            type_of_second_fixed_surface = msg.typeOfSecondFixedSurface
242            if hasattr(type_of_second_fixed_surface, "value"):
243                tosfs_val = type_of_second_fixed_surface.value
244            else:
245                tosfs_val = type_of_second_fixed_surface
246
247            # Group key: shortName + surface type + pdtn + second surface type
248            group_key = (var_name, int(toffs_val), int(pdtn_val), int(tosfs_val))
249
250            entry = _MsgEntry(
251                msg=msg,
252                file_uri=file_uri,
253                sec7_offset=sec7_offset,
254                sec7_length=sec7_length,
255                sec6_offset=sec6_offset,
256                sec6_length=sec6_length,
257                bmapflag=bmapflag,
258                index_section3=index["section3"][i],
259                index_section5=index["section5"][i],
260            )
261
262            all_var_messages.setdefault(group_key, []).append(entry)
263
264    def _matches_filters(self, msg) -> bool:
265        """Check if a message matches all user-supplied filters."""
266        for key, value in self.filters.items():
267            msg_val = getattr(msg, key, None)
268            if msg_val is None:
269                return False
270            # Handle Grib2Metadata objects
271            if hasattr(msg_val, "value"):
272                msg_val = msg_val.value
273            if msg_val != value:
274                return False
275        return True

281    def _build_variable_refs(
282        self,
283        var_name: str,
284        msg_entries: list,
285        refs: Dict[str, Any],
286    ) -> None:
287        """Build all Zarr refs for a single variable."""
288        # Map messages to dimensions
289        dim_mapping = _map_messages_to_dimensions(msg_entries)
290
291        dim_names = dim_mapping["dim_names"]  # ordered list of dim names
292        dim_values = dim_mapping["dim_values"]  # dict: dim_name -> sorted unique values
293        msg_index_map = dim_mapping["msg_index_map"]  # dict: dim_tuple -> msg_entry index
294
295        # Representative message for metadata
296        rep_msg = msg_entries[0].msg
297
298        # Compute shape (ensure plain Python ints for JSON serialization)
299        shape = [len(dim_values[d]) for d in dim_names] + [int(rep_msg.ny), int(rep_msg.nx)]
300        chunks = [1] * len(dim_names) + [int(rep_msg.ny), int(rep_msg.nx)]
301
302        # Build .zarray
303        codec_config = _build_codec_config(msg_entries[0])
304        zarray = _build_zarray_metadata(rep_msg, shape, chunks, codec_config)
305        refs[f"{var_name}/.zarray"] = json.dumps(zarray)
306
307        # Build .zattrs
308        dim_labels = dim_names + ["y", "x"]
309        zattrs = _build_zattrs(rep_msg, dim_labels)
310        refs[f"{var_name}/.zattrs"] = json.dumps(zattrs)
311
312        # Build data chunk refs
313        for dim_tuple, entry_idx in msg_index_map.items():
314            entry = msg_entries[entry_idx]
315            dim_indices = []
316            for i, d in enumerate(dim_names):
317                val = dim_tuple[i]
318                idx = list(dim_values[d]).index(val)
319                dim_indices.append(idx)
320
321            chunk_key = _build_chunk_key(var_name, dim_indices)
322
323            # The reference points to section 7 bytes in the original file.
324            # If bitmap is present, we include bitmap bytes as a companion ref
325            # and adjust the codec config per-chunk if needed.
326            if entry.bmapflag in {0, 254} and entry.sec6_offset is not None:
327                # For bitmap messages, we create a combined reference:
328                # The codec expects bitmap bytes prepended to section 7 bytes.
329                # We store them as separate companion refs and the main ref
330                # points to section 7 only. The codec config carries bitmap info.
331                refs[chunk_key] = [
332                    entry.file_uri,
333                    entry.sec7_offset,
334                    entry.sec7_length,
335                ]
336                # Store bitmap as companion reference
337                bitmap_key = f"{var_name}/.bitmap/{'.'.join(str(i) for i in dim_indices)}"
338                refs[bitmap_key] = [
339                    entry.file_uri,
340                    entry.sec6_offset,
341                    entry.sec6_length,
342                ]
343            else:
344                refs[chunk_key] = [
345                    entry.file_uri,
346                    entry.sec7_offset,
347                    entry.sec7_length,
348                ]
349
350        # Build coordinate arrays as inline base64-encoded refs
351        for dim_name in dim_names:
352            values = dim_values[dim_name]
353            _build_coord_refs(dim_name, values, refs)
354

420def _build_chunk_key(var_name: str, dim_indices: List[int]) -> str:
421    """Construct a Zarr chunk key like ``"TMP/0.0.0"``.
422
423    Parameters
424    ----------
425    var_name : str
426        Variable name (top-level Zarr array name).
427    dim_indices : list of int
428        Integer indices along each non-spatial dimension.
429
430    Returns
431    -------
432    str
433        Zarr chunk key, e.g. ``"TMP/0.1.0.0"`` where the trailing
434        two zeros are for the y and x spatial dimensions (always 0
435        since each message is one full grid).
436    """
437    parts = [str(i) for i in dim_indices] + ["0", "0"]
438    return f"{var_name}/{'.'.join(parts)}"

441def _build_zarray_metadata(
442    msg,
443    shape: List[int],
444    chunks: List[int],
445    codec_config: dict,
446) -> dict:
447    """Build ``.zarray`` JSON metadata for a variable.
448
449    Parameters
450    ----------
451    msg : Grib2Message
452        Representative message for dtype info.
453    shape : list of int
454        Full array shape including spatial dims.
455    chunks : list of int
456        Chunk shape (one message per chunk).
457    codec_config : dict
458        ``Grib2Codec`` configuration dict.
459
460    Returns
461    -------
462    dict
463        Zarr ``.zarray`` metadata.
464    """
465    dtype = "<f4" if msg.typeOfValues == 0 else "<i4"
466
467    return {
468        "zarr_format": 2,
469        "shape": shape,
470        "chunks": chunks,
471        "dtype": dtype,
472        "fill_value": "NaN" if dtype == "<f4" else 0,
473        "order": "C",
474        "compressor": codec_config,
475        "filters": None,
476    }

479def _build_zattrs(msg, dim_labels: List[str]) -> dict:
480    """Extract GRIB2 section metadata as Zarr attributes.
481
482    Parameters
483    ----------
484    msg : Grib2Message
485        Representative message.
486    dim_labels : list of str
487        Ordered dimension names including ``"y"`` and ``"x"``.
488
489    Returns
490    -------
491    dict
492        Zarr ``.zattrs`` metadata.
493    """
494    # Extract typeOfFirstFixedSurface - handle Grib2Metadata objects
495    type_of_first_fixed_surface = msg.typeOfFirstFixedSurface
496    if hasattr(type_of_first_fixed_surface, "value"):
497        type_of_first_fixed_surface = type_of_first_fixed_surface.value
498
499    # Extract valueOfFirstFixedSurface
500    value_of_first_fixed_surface = msg.valueOfFirstFixedSurface
501    if hasattr(value_of_first_fixed_surface, "value"):
502        value_of_first_fixed_surface = value_of_first_fixed_surface.value
503
504    # Extract refDate and leadTime
505    ref_date = msg.refDate
506    if isinstance(ref_date, np.datetime64):
507        ref_date = str(ref_date)
508    elif hasattr(ref_date, "isoformat"):
509        ref_date = ref_date.isoformat()
510    else:
511        ref_date = str(ref_date)
512
513    lead_time = msg.leadTime
514    if hasattr(lead_time, "total_seconds"):
515        lead_time = lead_time.total_seconds()
516    else:
517        lead_time = str(lead_time)
518
519    return {
520        "_ARRAY_DIMENSIONS": dim_labels,
521        "discipline": int(msg.section0[2]),
522        "parameterCategory": int(msg.parameterCategory),
523        "parameterNumber": int(msg.parameterNumber),
524        "typeOfFirstFixedSurface": int(type_of_first_fixed_surface),
525        "valueOfFirstFixedSurface": float(value_of_first_fixed_surface),
526        "refDate": ref_date,
527        "leadTime": lead_time,
528        "shortName": str(msg.shortName),
529        "fullName": str(msg.fullName),
530        "units": str(msg.units),
531    }

534def _build_codec_config(entry: _MsgEntry) -> dict:
535    """Build ``Grib2Codec`` configuration from a message entry.
536
537    Parameters
538    ----------
539    entry : _MsgEntry
540        Scanned message entry with index metadata.
541
542    Returns
543    -------
544    dict
545        Codec configuration suitable for ``Grib2Codec.from_config()``.
546    """
547    msg = entry.msg
548    sec3 = entry.index_section3
549    sec5 = entry.index_section5
550
551    # GDS: first 5 elements of section 3
552    gds = [int(x) for x in sec3[:5]]
553    # GDT: remaining elements of section 3
554    gdt = [int(x) for x in sec3[5:]]
555    # DRT number and template
556    drtn = int(sec5[1])
557    drt = [int(x) for x in sec5[2:]]
558
559    # Grid dimensions
560    nx = int(msg.nx)
561    ny = int(msg.ny)
562
563    # Scan mode flags
564    scan_mode_flags = None
565    if hasattr(msg, "scanModeFlags"):
566        scan_mode_flags = [int(x) for x in msg.scanModeFlags]
567
568    # Bitmap info
569    bitmap_flag = int(entry.bmapflag)
570    bitmap_offset = None
571    bitmap_length = None
572    if bitmap_flag in {0, 254} and entry.sec6_offset is not None:
573        bitmap_offset = int(entry.sec6_offset)
574        bitmap_length = int(entry.sec6_length)
575
576    # Number of data points and packed values
577    number_of_data_points = int(msg.numberOfDataPoints)
578    number_of_packed_values = int(msg.numberOfPackedValues)
579
580    # Type of values
581    type_of_values = int(msg.typeOfValues) if hasattr(msg, "typeOfValues") else 0
582
583    return {
584        "id": "grib2io",
585        "drtn": drtn,
586        "drt": drt,
587        "gdtn": int(msg.gdtn),
588        "gdt": gdt,
589        "gds": gds,
590        "nx": nx,
591        "ny": ny,
592        "bitmap_flag": bitmap_flag,
593        "bitmap_offset": bitmap_offset,
594        "bitmap_length": bitmap_length,
595        "scan_mode_flags": scan_mode_flags,
596        "type_of_values": type_of_values,
597        "number_of_data_points": number_of_data_points,
598        "number_of_packed_values": number_of_packed_values,
599    }

602def _get_dim_value(msg, dim_name: str) -> Any:
603    """Extract a dimension coordinate value from a message.
604
605    Parameters
606    ----------
607    msg : Grib2Message
608        The GRIB2 message.
609    dim_name : str
610        Dimension name.
611
612    Returns
613    -------
614    Any
615        The coordinate value, converted to a hashable/sortable type.
616    """
617    if dim_name == "level":
618        # Use the tuple (valueOfFirstFixedSurface, valueOfSecondFixedSurface)
619        # as the level identifier, matching xarray_backend logic
620        v1 = msg.valueOfFirstFixedSurface
621        v2 = msg.valueOfSecondFixedSurface
622        return (float(v1), float(v2))
623    elif dim_name == "refDate":
624        rd = msg.refDate
625        if hasattr(rd, "isoformat"):
626            return rd.isoformat()
627        return str(rd)
628    elif dim_name == "leadTime":
629        lt = msg.leadTime
630        if hasattr(lt, "total_seconds"):
631            return lt.total_seconds()
632        return str(lt)
633    elif dim_name == "duration":
634        d = msg.duration
635        if hasattr(d, "total_seconds"):
636            return d.total_seconds()
637        return str(d)
638    else:
639        val = getattr(msg, dim_name, None)
640        if hasattr(val, "value"):
641            val = val.value
642        return val

645def _map_messages_to_dimensions(
646    msg_entries: List[_MsgEntry],
647) -> dict:
648    """Group messages by variable and map to dimension indices.
649
650    This mirrors the logic in ``parse_grib_index()`` from the xarray
651    backend: for each message, extract the values of potential dimension
652    coordinates (level, leadTime, refDate, perturbationNumber, etc.),
653    determine which dimensions have more than one unique value, and
654    build a mapping from dimension-value tuples to message indices.
655
656    Parameters
657    ----------
658    msg_entries : list of _MsgEntry
659        All message entries for a single variable.
660
661    Returns
662    -------
663    dict
664        Dictionary with keys:
665        - ``"dim_names"``: ordered list of active dimension names
666        - ``"dim_values"``: dict mapping dim name to sorted unique values
667        - ``"msg_index_map"``: dict mapping dim-value tuple to entry index
668    """
669    # Collect dimension values for each message
670    all_dim_vals: Dict[str, list] = {d: [] for d in _ORDERED_DIM_NAMES}
671
672    for entry in msg_entries:
673        msg = entry.msg
674        for dim_name in _ORDERED_DIM_NAMES:
675            try:
676                val = _get_dim_value(msg, dim_name)
677                all_dim_vals[dim_name].append(val)
678            except (AttributeError, TypeError):
679                all_dim_vals[dim_name].append(None)
680
681    # Determine which dimensions are active (have >1 unique value)
682    active_dims = []
683    dim_values: Dict[str, list] = {}
684
685    for dim_name in _ORDERED_DIM_NAMES:
686        vals = all_dim_vals[dim_name]
687        # Filter out None values
688        non_none = [v for v in vals if v is not None]
689        if not non_none:
690            continue
691        unique_vals = sorted(set(non_none))
692        if len(unique_vals) > 1:

694            dim_values[dim_name] = unique_vals
695        elif len(unique_vals) == 1:

697            # to ensure every message gets a chunk key
698            active_dims.append(dim_name)
699            dim_values[dim_name] = unique_vals
700
701    # If no dimensions are active (single message), create a trivial mapping
702    if not active_dims:
703        # Use refDate as a single dimension of size 1
704        msg = msg_entries[0].msg
705        rd = _get_dim_value(msg, "refDate")
706        active_dims = ["refDate"]
707        dim_values = {"refDate": [rd]}
708
709    # Build the mapping from dimension-value tuples to entry indices
710    msg_index_map: Dict[tuple, int] = {}
711    for idx, entry in enumerate(msg_entries):
712        msg = entry.msg
713        dim_tuple = tuple(_get_dim_value(msg, d) for d in active_dims)
714        msg_index_map[dim_tuple] = idx

717        "dim_names": active_dims,
718        "dim_values": dim_values,

764        "zarr_format": 2,
765        "shape": [len(values)],
766        "chunks": [len(values)],

768        "fill_value": None if coord_values.dtype.kind in {"U", "S"} else 0,
769        "order": "C",
770        "compressor": None,
771        "filters": None,

780    refs[f"{dim_name}/0"] = "base64:" + b64_data

191    def _scan_file(
192        self,
193        file_path: str,
194        file_uri: str,
195        all_var_messages: Dict[str, list],
196    ) -> None:
197        """Scan a single GRIB2 file and collect message entries."""
198        with grib2io.open(file_path, save_index=False, use_index=False) as f:
199            index = f._index
200            msgs = list(f)
201
202        n_msgs = len(msgs)
203        for i in range(n_msgs):
204            msg = msgs[i]
205
206            # Apply filters
207            if not self._matches_filters(msg):
208                continue
209
210            sec_offsets = index["sectionOffset"][i]
211            sec_sizes = index["sectionSize"][i]
212            bmapflag = index["bmapflag"][i]
213
214            # Section 7 offset and length
215            sec7_offset = sec_offsets[7]
216            sec7_length = sec_sizes[7]
217
218            # Section 6 (bitmap) offset and length
219            sec6_offset = sec_offsets[6] if bmapflag in {0, 254} else None
220            sec6_length = sec_sizes[6] if bmapflag in {0, 254} else None
221
222            # Build a composite variable key that includes the surface type
223            # to avoid grouping messages with different surface types together.
224            # This mirrors how the xarray backend requires filtering to a
225            # single typeOfFirstFixedSurface.
226            var_name = str(msg.shortName)
227            type_of_first_fixed_surface = msg.typeOfFirstFixedSurface
228            if hasattr(type_of_first_fixed_surface, "value"):
229                toffs_val = type_of_first_fixed_surface.value
230            else:
231                toffs_val = type_of_first_fixed_surface
232
233            # Also include typeOfGeneratingProcess and
234            # productDefinitionTemplateNumber to disambiguate further
235            # (same approach as xarray backend's required_uniques)
236            pdtn = msg.productDefinitionTemplateNumber
237            if hasattr(pdtn, "value"):
238                pdtn_val = pdtn.value
239            else:
240                pdtn_val = pdtn
241
242            type_of_second_fixed_surface = msg.typeOfSecondFixedSurface
243            if hasattr(type_of_second_fixed_surface, "value"):
244                tosfs_val = type_of_second_fixed_surface.value
245            else:
246                tosfs_val = type_of_second_fixed_surface
247
248            # Group key: shortName + surface type + pdtn + second surface type
249            group_key = (var_name, int(toffs_val), int(pdtn_val), int(tosfs_val))
250
251            entry = _MsgEntry(
252                msg=msg,
253                file_uri=file_uri,
254                sec7_offset=sec7_offset,
255                sec7_length=sec7_length,
256                sec6_offset=sec6_offset,
257                sec6_length=sec6_length,
258                bmapflag=bmapflag,
259                index_section3=index["section3"][i],
260                index_section5=index["section5"][i],
261            )
262
263            all_var_messages.setdefault(group_key, []).append(entry)
264
265    def _matches_filters(self, msg) -> bool:
266        """Check if a message matches all user-supplied filters."""
267        for key, value in self.filters.items():
268            msg_val = getattr(msg, key, None)
269            if msg_val is None:
270                return False
271            # Handle Grib2Metadata objects
272            if hasattr(msg_val, "value"):
273                msg_val = msg_val.value
274            if msg_val != value:
275                return False
276        return True

329                # The codec expects bitmap bytes prepended to section 7 bytes.
330                # We store them as separate companion refs and the main ref

335                    entry.sec7_length,

337                # Store bitmap as companion reference

343                ]

349                ]
